101  elements_t Es(c.get<elements_t>("elements"));
102 
103  p_elements_t result;

106 
107  size_t idx=0;
108  for(elements_t::iterator it=Es.begin(), end=Es.end(); it!=end; ++it)

115  if(eit==p_info.elements.end())
116  throw key_error(etype);

119 
120  ElementVoid *E;
121  try{
122  E = builder->build(EC);
123  }catch(key_error& e){

126  <<"' : missing required parameter '"<<e.what()<<"'";

128 
129  }catch(std::exception& e){

136  if(E->type_name()!=etype) {

166 {

168 
169  S->next_elem = start;
