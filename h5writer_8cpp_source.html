65 void H5StateWriter::open(const std::string& spec)
66 {
67  try {
68  close();
69  /* The provided spec may contain both file path and group(s)
70  * seperated by '/' which is ambigious as the file path
71  * may contain '/' as well...
72  * so do as h5ls does and strip off from the right hand side until
73  * and try to open while '/' remain.
74  */
75  size_t sep = spec.npos;
76 
77  while(true) {
78  sep = spec.find_last_of('/', sep-1);
79 
80  std::string fname(spec.substr(0, sep));
81 
82  try {
83  pvt->file.openFile(fname, H5F_ACC_RDWR|H5F_ACC_CREAT);
84  } catch(H5::FileIException& e) {
85  if(sep==spec.npos) {
86  // no more '/' so this is failure
87  throw std::runtime_error("Unable to open file");
88  }
89  continue; // keep trying
90  } CATCH()
91 
92  if(sep!=spec.npos) {
93  std::string group(spec.substr(sep+1));
94  // delete group if it already exists
95  try{
96  H5G_stat_t stat;
97  pvt->file.getObjinfo(group, stat);
98  pvt->file.unlink(group);
99  } catch(H5::FileIException&) {
100  // ignore non-existant
101  }
102 
103  pvt->group = pvt->file.createGroup(group);
104  } else {
105  //TODO: cleanup root?
106 
107  pvt->group = pvt->file.openGroup("/");
108  }
109 
110  return;
111  }
112  } CATCH()
113 }

123 void H5StateWriter::prepare(const StateBase *RS)
124 {
125  try {
126  // hack since getArray() is non-const
127  // we won't actually modify the state
128  StateBase *S = const_cast<StateBase*>(RS);
129 
130  assert(pvt->elements.empty());
131 
132  for(unsigned idx=0; true; idx++) {
133  StateBase::ArrayInfo info;
134 
135  if(!S->getArray(idx, info))
136  break;
137 
138  H5::DataType dtype;
139  switch(info.type) {
140  case StateBase::ArrayInfo::Double:
141  dtype = H5::DataType(H5::PredType::NATIVE_DOUBLE);
142  break;
143  case StateBase::ArrayInfo::Sizet:
144  if(sizeof(size_t)==8)
145  dtype = H5::DataType(H5::PredType::NATIVE_UINT64);
146  else if(sizeof(size_t)==4)
147  dtype = H5::DataType(H5::PredType::NATIVE_UINT32);
148  else
149  throw std::logic_error("unsupported size_t");
150  break;
151  default:
152  continue; // TODO string
153  }
154 
155  StateElement elem;
156  elem.idx = idx;
157  elem.info = info;
158 
159  // first dim is simulation "time"
160  std::vector<hsize_t> dims(info.ndim+1), maxdims(info.ndim+1);
161  for(size_t i=0; i<info.ndim; i++) {
162  maxdims[i+1] = dims[i+1] = info.dim[i];
163  }
164 
165  elem.shape = dims; // copy
166 
167  // size w/ first dim==0
168  dims[0] = 0;
169  maxdims[0] = H5S_UNLIMITED;
170  H5::DataSpace dspace(dims.size(), &dims[0], &maxdims[0]);
171 
172  dims[0] = 10; // chunk size in "time" steps
173  H5::DSetCreatPropList props;
174  props.setChunk(dims.size(), &dims[0]);
175 
176  // memspace is simple from origin to [1,shape]
177  dims[0] = 1;
178  elem.memspace = H5::DataSpace(dims.size(), &dims[0]);
179 
180  elem.dset = pvt->group.createDataSet(info.name, dtype, dspace, props);
181 
182  pvt->elements.push_back(elem);
183  }
184 
185  if(pvt->elements.empty()) {
186  throw std::logic_error("state type has not elements to store?");
187  }
188  } CATCH()
189 }

191 void H5StateWriter::append(const StateBase *RS)
192 {
193  try {
194  StateBase *S = const_cast<StateBase*>(RS);
195 
196  if(pvt->elements.empty())
197  prepare(RS);
198 
199  BOOST_FOREACH(StateElement& elem, pvt->elements)
200  {
201  StateBase::ArrayInfo info;
202 
203  if(!S->getArray(elem.idx, info))
204  throw std::logic_error("can't re-fetch state parameter?");
205 
206  assert((elem.info.ndim==info.ndim) && (elem.info.type==info.type));
207 
208  size_t index = elem.shape[0]; // we will write data[index,...]
209  elem.shape[0]++;
210 
211  elem.dset.extend(&elem.shape[0]); // resize
212 
213  // filespace is hyper from [index,0...] to [index,shape]
214  std::vector<hsize_t> start(elem.shape.size(), 0);
215  start[0] = index;
216 
217  elem.shape[0] = 1; // reuse as count
218 
219  H5::DataSpace filespace(elem.dset.getSpace());
220  filespace.selectHyperslab(H5S_SELECT_SET, &elem.shape[0], &start[0]);
221 
222  H5::DataType dtype(elem.dset.getDataType());
223 
224  elem.dset.write(info.ptr, dtype, elem.memspace, filespace);
225 
226  elem.shape[0] = index+1;
227  }
228  } CATCH()
229 }
