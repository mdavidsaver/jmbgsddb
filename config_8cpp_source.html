204 void assign_expr_to_Config(Config& conf, const std::string& name, const expr_t& expr)
205 {
206  switch(expr.etype)
207  {
208  case glps_expr_number:
209  conf.set<double>(name, boost::get<double>(expr.value));
210  break;
211  case glps_expr_string:
212  conf.set<std::string>(name, boost::get<std::string>(expr.value));
213  break;
214  case glps_expr_vector:
215  conf.set<std::vector<double> >(name, boost::get<std::vector<double> >(expr.value));
216  break;
217  case glps_expr_config: {
218  boost::shared_ptr<Config> pconf(boost::get<boost::shared_ptr<Config> >(expr.value));
219  std::vector<Config> cvect;
220  if(pconf)
221  cvect.push_back(*pconf);
222  conf.set<std::vector<Config> >(name, cvect);
223  }
224  break;
225  default:
226  throw std::logic_error("Context contained unresolved/illegal variable");
227  }
228 }

238  void fill_vars(parse_context& ctxt)
239  {
240  for(values_t::const_iterator it=vars.begin(), end=vars.end(); it!=end; ++it)
241  {

247  Config* fill_context(parse_context& ctxt)
248  {
249  std::auto_ptr<Config> ret(new Config);
250  ret->reserve(ctxt.vars.size()+2);
251 
252  // copy ctxt.vars to top level Config
253  for(parse_context::vars_t::iterator it=ctxt.vars.begin(), end=ctxt.vars.end();
254  it!=end; ++it)
255  {
256  assign_expr_to_Config(*ret, it->name, it->expr);
257  }
258 
259  if(ctxt.line.size()==0)
260  throw std::runtime_error("No beamlines defined by this file");
261 
262  parse_line *line = NULL;
263 
264  {
265  // find the magic "USE" element. eg "USE: linename;"
266  parse_context::map_idx_t::const_iterator it=ctxt.element_idx.find("USE");
267  if(it!=ctxt.element_idx.end()) {
268  parse_element &elem = ctxt.elements[it->second];
269  parse_context::map_idx_t::const_iterator lit = ctxt.line_idx.find(elem.etype);
270 
271  if(lit!=ctxt.line_idx.end()) {
272  line = &ctxt.line[lit->second];
273  } else {
274  std::ostringstream strm;
275  strm<<"\"USE: "<<elem.etype<<";\" references undefined beamline";
276  throw std::runtime_error(strm.str());
277  }
278  } else {
279  // no magic USE, default to last line
280  line = &ctxt.line.back();
281  }
282  }
283 
284  assert(line);
285 
286  if(line->names.size()==0) {
287  std::ostringstream strm;
288  strm<<"Beamline '"<<line->label<<"' has no elements";
289  throw std::runtime_error(strm.str());
290  }
291 
292  Config::vector_t elements;
293  elements.resize(line->names.size());
294 
295  // copy in elements
296  size_t i = 0;
297  for(strlist_t::list_t::const_iterator it=line->names.begin(), end=line->names.end();
298  it!=end; ++it)
299  {
300  Config next(ret->new_scope()); // inhiert global scope
301  const parse_element& elem = ctxt.elements[ctxt.element_idx[*it]];
302 
303  next.reserve(elem.props.size()+2);
304 
305  // push elements properties
306  for(kvlist_t::map_t::const_iterator itx=elem.props.begin(), endx=elem.props.end();
307  itx!=endx; ++itx)
308  {
309  assign_expr_to_Config(next, itx->first, itx->second);
310  }
311 
312  // special properties
313  assert(!elem.etype.empty() && !elem.label.empty());
314  next.set<std::string>("type", elem.etype);
315  next.set<std::string>("name", elem.label);
316  elements[i++].swap(next);
317  }
318 
319  ret->swap<std::string>("name", line->label);
320  ret->swap<Config::vector_t>("elements", elements);
321 
322  return ret.release();
323  }
324 };

480 void GLPSPrint(std::ostream& strm, const Config& conf)
481 {
482  // print variables
483  for(Config::const_iterator it=conf.begin(), end=conf.end();
484  it!=end; ++it)
485  {
486  boost::apply_visitor(glps_show(strm, it->first), it->second);
487  }
488 
489  const Config::vector_t *v;
490  try{
491  v = &conf.get<Config::vector_t>("elements");
492  }catch(key_error&){
493  strm<<"# Missing beamline element list\n";
494  return;
495  }catch(boost::bad_get&){
496  strm<<"# 'elements' is not a beamline element list\n";
497  return;
498  }
499 
500  std::vector<std::string> line;
501  line.reserve(v->size());
502 
503  std::set<std::string> eshown;
504 
505  // print element definitions
506  for(Config::vector_t::const_iterator it=v->begin(), end=v->end();
507  it!=end; ++it)
508  {
509  bool ok = true;
510  try {
511  const std::string& name=it->get<std::string>("name");
512  const std::string& type=it->get<std::string>("type");
513  if(name.empty() || type.empty())
514  throw std::runtime_error("Element missing 'name' and/or 'type'");
515  line.push_back(name);
516  // only show element definition once
517  if(eshown.find(name)!=eshown.end())
518  continue;
519  strm<<name<<": "<<type;
520  eshown.insert(name);
521  }catch(key_error&){
522  ok=false;
523  }catch(boost::bad_get&){
524  ok=false;
525  }
526  if(!ok)
527  strm<<"# <malformed element>";
528 
529  for(Config::const_iterator itx=it->begin(), endx=it->end();
530  itx!=endx; ++itx)
531  {
532  if(itx->first=="name" || itx->first=="type")

538  }
