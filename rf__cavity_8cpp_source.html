141 void calTransfac(const numeric_table& fldmap, int column_no, const double IonK, const bool half,
142  double &Ecenter, double &T, double &Tp, double &S, double &Sp, double &V0)
143 {
144  // Compute electric center, amplitude, and transit time factors [T, Tp, S, Sp] for RF cavity mode.
145  int n, k;
146  double dz, eml, em_mom;
147  std::vector<double> z, EM;
148 
149  column_no--; // F*** fortran
150  assert(column_no>0);
151 
152  n = fldmap.table.size1();
153  if (half) n = (int)round((n-1)/2e0);
154 
155  if(n<=0 || (size_t)column_no>=fldmap.table.size2())
156  throw std::runtime_error("field map size invalid");
157 
158  z.resize(n);
159  EM.resize(n);
160 
161  std::copy(fldmap.table.find1(2, 0, 0),
162  fldmap.table.find1(2, n, 0),
163  z.begin());
164  std::copy(fldmap.table.find1(2, 0, column_no),
165  fldmap.table.find1(2, n, column_no),
166  EM.begin());
167 
168  dz = (z[n-1]-z[0])/(n-1);
169 
170 // prt_data(z, EM);
171 
172  // Start at zero.
173  for (k = n-1; k >= 0; k--)
174  z[k] -= z[0];
175 
176  eml = 0e0, em_mom = 0e0;
177  for (k = 0; k < n-1; k++) {
178  eml += (fabs(EM[k])+fabs(EM[k+1]))/2e0*dz;
179  em_mom += (z[k]+z[k+1])/2e0*(fabs(EM[k])+fabs(EM[k+1]))/2e0*dz;
180  }
181  Ecenter = em_mom/eml;
182 
183  for (k = 0; k < n; k++)
184  z[k] -= Ecenter;
185 
186  T = 0;
187  for (k = 0; k < n-1; k++)
188  T += (EM[k]+EM[k+1])/2e0*cos(IonK*(z[k]+z[k+1])/2e0)*dz;
189  T /= eml;
190 
191  Tp = 0;
192  for (k = 0; k < n-1; k++)
193  Tp -= ((z[k]+z[k+1])/2e0)*(EM[k]+EM[k+1])/2e0*sin(IonK*(z[k]+z[k+1])/2e0)*dz;
194  Tp /= eml;
195 
196  S = 0e0;
197  for (k = 0; k < n-1; k++)
198  S += (EM[k]+EM[k+1])/2e0*sin(IonK*(z[k]+z[k+1])/2e0)*(z[k+1]-z[k]);
199  S /= eml;
200 
201  Sp = 0e0;
202  for (k = 0; k < n-1; k++) {
203  Sp += (z[k]+z[k+1])/2e0*(EM[k]+EM[k+1])/2e0*cos(IonK*(z[k]+z[k+1])/2e0)*dz;
204  }
205  Sp /= eml;
206 
207  V0 = eml/MeVtoeV/MtoMM;
208 }

649 static
650 void EvalGapModel(const double dis, const double IonW0, const Particle &real, const double IonFy0,
651  const double k, const double Lambda, const double Ecen,
652  const double T, const double S, const double Tp, const double Sp, const double V0,
653  double &IonW_f, double &IonFy_f)
654 {
655  double Iongamma_f, IonBeta_f, k_f;
656 
657  IonW_f = IonW0 + real.IonZ*V0*T*cos(IonFy0+k*Ecen)*MeVtoeV - real.IonZ*V0*S*sin(IonFy0+k*Ecen)*MeVtoeV;
658  Iongamma_f = IonW_f/real.IonEs;
659  IonBeta_f = sqrt(1e0-1e0/sqr(Iongamma_f));
660  k_f = 2e0*M_PI/(IonBeta_f*Lambda);
661 
662  IonFy_f = IonFy0 + k*Ecen + k_f*(dis-Ecen)
663  + real.IonZ*V0*k*(Tp*sin(IonFy0+k*Ecen)+Sp*cos(IonFy0+k*Ecen))/(2e0*(IonW0-real.IonEs)/MeVtoeV);
664 }
