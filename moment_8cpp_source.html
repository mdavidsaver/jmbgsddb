26 bool load_storage(ARR& to, const Config& conf, const std::string& name, bool T=true)
27 {
28  try{
29  const std::vector<double>& val(conf.get<std::vector<double> >(name));
30  if(to.size()!=val.size()) {
31  std::ostringstream strm;
32  strm<<"Array "<<name<<" must have "<<to.size()<<" elements, not "<<val.size();
33  throw std::invalid_argument(strm.str());
34  }
35  std::copy(val.begin(), val.end(), to.begin());
36 
37  return true;
38  }catch(key_error&){
39  if(T)
40  throw std::invalid_argument(name+" not defined");
41  else
42  return false;
43  // default to identity
44  }catch(boost::bad_any_cast&){
45  if(T)
46  throw std::invalid_argument(name+" has wrong type (must be vector)");
47  else
48  return false;
49  }
50 }

83  real.reserve(10);

87  size_t icstate = (size_t)icstate_f;
88 
89  std::string vectorname(c.get<std::string>("vector_variable", "moment0"));
90  std::string matrixname(c.get<std::string>("matrix_variable", "initial"));
91 
92  std::vector<double> ics, nchg;
93  bool have_ics = c.tryGet<std::vector<double> >("IonChargeStates", ics);
94 
95  ref.IonEs = c.get<double>("IonEs", 0e0),
96  ref.IonEk = c.get<double>("IonEk", 0e0);
97  ref.recalc();
98 
99  if(!have_ics) {
100  ref.IonZ = c.get<double>("IonZ", 0e0);
101  ref.IonQ = c.get<double>("IonQ", 1e0);
102 
103  ics.push_back(ref.IonZ);
104  nchg.push_back(ref.IonQ);
105 
106  } else {
107  if(ics.empty())
108  throw std::invalid_argument("IonChargeStates w/ length 0");
109  if(icstate>=ics.size())
110  throw std::invalid_argument("IonChargeStates[cstate] is out of bounds");
111 
112  nchg = c.get<std::vector<double> >("NCharge");
113  if(nchg.size()!=ics.size())
114  throw std::invalid_argument("NCharge[] and IonChargeStates[] must have equal length");
115 
116  ref.IonZ = ics[0];
117  ref.IonQ = nchg[0];
118  }
119 
120  /* Possible configurations
121  * 1. Neither 'cstate' nor 'IonChargeStates' defined (empty Config).
122  * No charge states, must go through source element to be useful
123  * 2. 'IonChargeStates' defined, but not 'cstate'.
124  * Load all charge states
125  * 3. 'cstate' and 'IonChargeStates' defined.
126  * Load a single charge state
127  */

129  // no-op
130  } else if(!have_cstate && have_ics) {
131  // many charge states
132  icstate = 0;
133 
134  } else if(have_cstate && have_ics) {
135  // single charge state
136 
137  // drop other than selected state
138  ics[0] = ics[icstate];
139  nchg[0] = nchg[icstate];
140  ics.resize(1);
141  nchg.resize(1);
142 
143  } else {
144  throw std::invalid_argument("MomentState: must define IonChargeStates and NCharge when cstate is set");
145  }
146 
147  if(have_ics) {
148  real.resize(ics.size());
149  moment0.resize(ics.size());
150  moment1.resize(ics.size());
151 
152  for(size_t i=0; i<ics.size(); i++) {
153  std::string num(boost::lexical_cast<std::string>(icstate+i));
154 
155  moment0[i].resize(maxsize);
156  moment1[i].resize(maxsize, maxsize);
157  moment1[i] = boost::numeric::ublas::identity_matrix<double>(maxsize);
158 
159  load_storage(moment0[i].data(), c, vectorname+num);
160  load_storage(moment1[i].data(), c, matrixname+num);
161 
162  real[i] = ref;
163 
164  real[i].IonZ = ics[i];
165  real[i].IonQ = nchg[i];
166 
167  real[i].phis = moment0[i][PS_S];
168  real[i].IonEk += moment0[i][PS_PS]*MeVtoeV;
169 
170  real[i].recalc();
171  }
172  } else {
173  real.resize(1); // hack, ensure at least one element so getArray() can return some pointer
174  real[0] = ref;
175 
176  moment0.resize(1);
177  moment1.resize(1);
178  moment0[0].resize(maxsize);
179  moment1[0].resize(maxsize, maxsize);
180 
181  load_storage(moment0[0].data(), c, vectorname, false);
182  load_storage(moment1[0].data(), c, matrixname, false);
183  }

190 void MomentState::calc_rms()
191 {
192  assert(real.size()>0);
193  assert(moment0_env.size()==maxsize);
194  assert(moment0_rms.size()==maxsize);
195  assert(moment1_env.size1()==maxsize);
196  assert(moment1_env.size2()==maxsize);
197 
198  double totQ = 0.0;
199  for(size_t n=0; n<real.size(); n++) {
200  totQ += real[n].IonQ;
201  if(n==0)
202  noalias(moment0_env) = moment0[n]*real[n].IonQ;
203  else
204  noalias(moment0_env) += moment0[n]*real[n].IonQ;
205  }
206  moment0_env /= totQ;
207 
208  for(size_t j=0; j<maxsize; j++) {
209  double variance = 0.0;
210  for(size_t n=0; n<moment0.size(); n++) {
211  const double Q = real[n].IonQ;
212  const double diff = moment0[n][j]-moment0_env[j];
213 
214  variance += Q*(moment1[n](j,j) + diff*diff); // Q * (sum squares + square of the sum)
215  }
216 
217  moment0_rms[j] = sqrt(variance/totQ);
218  }
219 
220  for(size_t n=0; n<real.size(); n++) {
221  if(n==0)
222  noalias(moment1_env) = moment1[n]*real[n].IonQ;
223  else
224  noalias(moment1_env) += moment1[n]*real[n].IonQ;
225  }
226  moment1_env /= totQ;
227 }

490 void MomentElementBase::get_misalign(const state_t &ST, const Particle &real, value_t &M, value_t &IM) const
491 {
492  state_t::matrix_t R, R_inv,
493  scl = boost::numeric::ublas::identity_matrix<double>(state_t::maxsize),
494  scl_inv = scl,
495  T = scl,
496  T_inv = scl;
497 
498  scl(state_t::PS_S, state_t::PS_S) /= -real.SampleIonK;
499  scl(state_t::PS_PS, state_t::PS_PS) /= sqr(real.beta)*real.gamma*ST.ref.IonEs/MeVtoeV;
500 
501  inverse(scl_inv, scl);
502 
503  // Translate to center of element.
504  T(state_t::PS_S, 6) = -length/2e0*MtoMM;
505  T(state_t::PS_PS, 6) = 1e0;
506  inverse(T_inv, T);
507 
508  RotMat(dx, dy, pitch, yaw, tilt, R);
509 
510  M = prod(T, scl);
511  M = prod(R, M);
512  M = prod(T_inv, M);
513  M = prod(scl_inv, M);
514 
515  // J.B. Bug in TLM: should be inverse.
516  RotMat(-dx, -dy, -pitch, -yaw, -tilt, R_inv);
517 
518  // Translate to center of element.
519  T(state_t::PS_S, 6) = length/2e0*MtoMM;
520  T(state_t::PS_PS, 6) = 1e0;
521  inverse(T_inv, T);
522 
523  IM = prod(T, scl);
524  IM = prod(R_inv, IM);
525  IM = prod(T_inv, IM);
526  IM = prod(scl_inv, IM);
527 }

529 void MomentElementBase::advance(StateBase& s)
530 {
531  state_t& ST = static_cast<state_t&>(s);
532  using namespace boost::numeric::ublas;
533 
534  // IonEk is Es + E_state; the latter is set by user.
535  ST.recalc();
536 
537  if(!check_cache(ST))
538  {
539  // need to re-calculate energy dependent terms
540 
541  resize_cache(ST);
542 
543  recompute_matrix(ST); // updates transfer and last_Kenergy_out
544 
545  ST.recalc();
546  }
547 
548  // recompute_matrix only called when ST.IonEk != last_Kenergy_in.
549  // Matrix elements are scaled with particle energy.
550 
551  ST.pos += length;
552 
553  ST.ref.phis += ST.ref.SampleIonK*length*MtoMM;
554 
555  for(size_t k=0; k<last_Kenergy_in.size(); k++) {
556  ST.real[k].phis += ST.real[k].SampleIonK*length*MtoMM;
557  ST.real[k].IonEk = last_Kenergy_out[k];
558 
559  ST.moment0[k] = prod(transfer[k], ST.moment0[k]);
560 
561  scratch = prod(transfer[k], ST.moment1[k]);
562  ST.moment1[k] = prod(scratch, trans(transfer[k]));
563  }
564 
565  ST.calc_rms();
566 }

568 bool MomentElementBase::check_cache(const state_t& ST) const
569 {
570  if(skipcache) return false;
571  if(last_Kenergy_in.size()!=ST.size()) return false; // different # of charge states
572 
573  for(size_t k=0; k<last_Kenergy_in.size(); k++) {
574  if(last_Kenergy_in[k]!=ST.real[k].IonEk) return false;
575  }
576  return true;
