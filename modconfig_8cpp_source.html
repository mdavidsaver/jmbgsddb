22 void Dict2Config(Config& ret, PyObject *dict, unsigned depth)
23 {
24  if(depth>3)
25  throw std::runtime_error("too deep for Dict2Config");
26 
27  PyObject *key, *value;
28  Py_ssize_t pos = 0;
29 
30  while(PyDict_Next(dict, &pos, &key, &value)) {
31  PyRef<> keyref(key, borrow());
32  PyCString skey(keyref);
33  const char *kname = skey.c_str();
34 
35  PyTypeObject *valuetype = (PyTypeObject*)PyObject_Type(value);
36  if(valuetype==&PyFloat_Type) { // scalar double
37  double val = PyFloat_AsDouble(value);
38  ret.set<double>(kname, val);
39 
40  } else if(valuetype==&PyInt_Type) { // scalar integer (treated as double)
41  long val = PyInt_AsLong(value);
42  ret.set<double>(kname, val);
43 
44  } else if(PyString_Check(value)) { // string
45  PyRef<> valref(value, borrow());
46  PyCString sval(valref);
47  const char *val = sval.c_str();
48 
49  ret.set<std::string>(kname, val);
50 
51  } else if(PyArray_Check(value)) { // array (ndarray)
52  PyRef<> arr(PyArray_ContiguousFromAny(value, NPY_DOUBLE, 0, 2));
53  double *buf = (double*)PyArray_DATA(arr.py());
54  std::vector<double> temp(PyArray_SIZE(arr.py()));
55  std::copy(buf, buf+temp.size(), temp.begin());
56 
57  ret.swap<std::vector<double> >(kname, temp);
58 
59  } else if(PySequence_Check(value)) { // list of dict
60  Py_ssize_t N = PySequence_Size(value);
61 
62  Config::vector_t output;
63  output.reserve(N);
64 
65  for(Py_ssize_t i=0; i<N; i++) {
66  PyObject *elem = PySequence_GetItem(value, i);
67  assert(elem);
68 
69  if(!PyDict_Check(elem))
70  throw std::invalid_argument("lists must contain only dict()s");
71 
72  //output.push_back(ret.new_scope()); // TODO: can't use scoping here since iteration order of PyDict_Next() is not stable
73  output.push_back(Config());
74 
75  Dict2Config(output.back(), elem, depth+1); // inheirt parent scope
76  }
77 
78  ret.set<Config::vector_t>(kname, output);
79 
80  } else {
81  std::ostringstream msg;
82  msg<<"Must be a dict, not "<<valuetype->tp_name;
83  throw std::invalid_argument(msg.str());
84  }
85  }

187  // no-op

193  PyRef<> keyx(key, borrow());

196  Config::value_t curval;
197

201 
202  } else if(PyString_Check(value)) {
203  PyRef<> valuex(value, borrow());
204  PyCString valstr(valuex);
205 
206  curval = valstr.c_str();
207 
208  } else {
209  PyErr_SetString(PyExc_ValueError, "extra {} can contain only numbers or strings");
210  return NULL;
211  }
212 
213  parser.setVar(keystr.c_str(), curval);
214  }
215  } else {
216  PyErr_SetString(PyExc_ValueError, "'extra' must be a dict");
217  return NULL;
218  }
219 
220  PyGetBuf buf;

222 
223  if(PyDict_Check(conf)) {
224  C.reset(dict2conf(conf));
225
