154 PyObject *PyMachine_propagate(PyObject *raw, PyObject *args, PyObject *kws)
155 {
156 
157  TRY {
158  PyObject *state, *toobserv = Py_None;
159  unsigned long start = 0, max = (unsigned long)-1;
160  const char *pnames[] = {"state", "start", "max", "observe", NULL};
161  if(!PyArg_ParseTupleAndKeywords(args, kws, "O|kkO", (char**)pnames, &state, &start, &max, &toobserv))
162  return NULL;
163 
164  PyStoreObserver observer;
165  PyScopedObserver observing(machine->machine);
166 
167  if(toobserv!=Py_None) {
168  PyRef<> iter(PyObject_GetIter(toobserv)), item;
169 
170  while(item.reset(PyIter_Next(iter.py()), PyRef<>::allow_null())) {
171  Py_ssize_t num = PyNumber_AsSsize_t(item.py(), PyExc_ValueError);
172  if(PyErr_Occurred())
173  throw std::runtime_error(""); // caller will get active python exception
174 
175  observing.observe(num, &observer);
176 
177  }
178  }
179 
180  machine->machine->propagate(unwrapstate(state), start, max);
181  if(toobserv) {
182  return observer.list.release();
183  } else {
184  Py_RETURN_NONE;
185  }
186  } CATCH2(std::invalid_argument, ValueError)
187  CATCH()
